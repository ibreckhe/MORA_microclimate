
    model {
    ## Priors
    alpha ~ dnorm(0, .001)
    beta ~ dnorm(0, .001)
    sdy ~ dnorm(pop_sdy,0.1)T(0,)
    tauy <- 1 / (sdy * sdy)
    sdx ~ dnorm(pop_sdx,0.1)T(0,)
    taux <- 1 / (sdx * sdx)
    
    ## Likelihood
    for (i in 1:n){
    xtrue[i] ~ dnorm(0,taux)
    x[i] ~ dnorm(xtrue[i], obs_taux)
    y[i] ~ dnorm(ytrue[i], obs_tauy)
    ytrue[i] ~ dnorm(mu[i],tauy)
    mu[i] <- alpha + beta * xtrue[i]
    }
    }
